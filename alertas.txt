<script type="text/javascript">


$('.AlertDelete').on("click", function(e) {
  e.preventDefault();
  var url = $(this).attr('href');
  swal({
      html:true,
      title: "<h3>¿Quieres eliminar este registro?</h3>",
      text: "No podrá recuperar el registro una vez sea eliminado!",
      type: "warning",
      showCancelButton: true,
      confirmButtonColor: '#DD6B55',
      confirmButtonText: 'Si, Eliminarlo!',
      cancelButtonText: "No, Cancelar!",
      confirmButtonClass: "btn-danger",
      closeOnConfirm: false,
      closeOnCancel: false
    },
    function(isConfirm) {
      if (isConfirm) {
        swal("Eliminado!", "La categoria ha sido eliminado!", "error");
          window.location.replace(url);
      } else {
        swal("Cancelado", "Su cliente está a salvo! :)", "success");
      }
    });
});

</script>
